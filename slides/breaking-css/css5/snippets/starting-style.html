div {
    /* No background color declared */
    display: none;
    transition: background-color 3s;
}
    
div.showing {
    /* The background color which will be used when the showing class is inserted and after 3 seconds of delay. */
    background-color: skyblue;
    display: block; /* The element will be rendered then @starting-style will be called */
}
    
@starting-style {
    div.showing {
        /*The background color which will be used immediatelly when the showing class is inserted. */
        background-color: red;
    }
}
